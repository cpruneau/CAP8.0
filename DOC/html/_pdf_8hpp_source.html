142inline double fDistributionPdf(double x, double n, double m, double x0 = 0)
143{
144 // function is defined only for both n and m > 0
145 if (n < 0 || m < 0) throw MathException("fDistributionPdf() called with n < 0 || m < 0");
146 if ((x-x0) < 0) return 0.0;
147 return std::exp((n/2) * std::log(n) + (m/2) * std::log(m) + lgamma((n+m)/2) - lgamma(n/2) - lgamma(m/2)
148 + (n/2 -1) * std::log(x-x0) - ((n+m)/2) * std::log(m + n*(x-x0)) );
149
150}

161inline double gammaPdf(double x, double alpha, double theta, double x0 = 0)
162{
163 // Inlined to enable clad-auto-derivation for this function.
164
165 if ((x-x0) < 0) throw MathException("gammaPdf() called with (x-x0) < 0");
166 if ((x-x0) == 0) return (alpha==1) ? 1.0/theta : 0.0;
167 return (alpha == 1) ?
168 std::exp(-(x-x0)/theta)/theta :
169 std::exp((alpha - 1) * std::log((x-x0)/theta) - (x-x0)/theta - lgamma(alpha))/theta;
170}

209inline double biGaussianPdf(double x, double y, double sigmax = 1, double sigmay = 1, double rho = 0, double x0 = 0, double y0 = 0)
210{
211 double u = (x-x0)/sigmax;
212 double v = (y-y0)/sigmay;
213 double c = 1. - rho*rho;
214 double z = u*u - 2.*rho*u*v + v*v;
215 double norm = twoPi() * sigmax * sigmay * std::sqrt(c);
216 return std::exp(-z/(2.0*c))/norm;
217}
