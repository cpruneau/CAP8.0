270 virtual void configureObjects(const Configuration & configuration)
271 {
272 if (reportDebug(__FUNCTION__)) { printCR(); }
273 String name;
274 String title;
275 bool owner;
276 String importPath;
277 String exportPath;
278 unsigned int nObjects;
279 _use = configuration.valueBool( createKey(_taskName,_typeName,"USE"));
280 _create = configuration.valueBool( createKey(_taskName,_typeName,"CREATE"));
281 _import = configuration.valueBool( createKey(_taskName,_typeName,"IMPORT"));
282 _importBaseName = configuration.valueString(createKey(_taskName,_typeName,"IMPORT:BASE_NAME"));
283 importPath = configuration.valueString(createKey(_taskName,_typeName,"IMPORT:PATH"));
284 _importFileName = configuration.valueString(createKey(_taskName,_typeName,"IMPORT:FILE_NAME"));
285 _export = configuration.valueBool( createKey(_taskName,_typeName,"EXPORT"));
286 _exportBaseName = configuration.valueString(createKey(_taskName,_typeName,"EXPORT:BASE_NAME"));
287 exportPath = configuration.valueString(createKey(_taskName,_typeName,"EXPORT:PATH"));
288 _exportFileName = configuration.valueString(createKey(_taskName,_typeName,"EXPORT:FILE_NAME"));
289 _forceRewrite = configuration.valueBool( createKey(_taskName,_typeName,"EXPORT:FORCE_REWRITE"));
290 nObjects = configuration.valueInt( createKey(_taskName,_typeName,"N") );
291
292 EnvironmentVariables * envVariables = EnvironmentVariables::environmentVariables();
293 String basePath;
294
295 if (_typeName.BeginsWith("HISTOGRAM"))
296 {
297 if (importPath.EqualTo("DEFAULT"))
298 {
299 _importPath = envVariables->variable("CAP_HISTOS_IMPORT_PATH");
300 }
301 else if (importPath.BeginsWith("/"))
302 {
303 _importPath = importPath;
304 }
305 else
306 {
307 _importPath = envVariables->variable("CAP_HISTOS_IMPORT_PATH");
308 _importPath += "/";
309 _importPath += importPath;
310 }
311 if (exportPath.EqualTo("DEFAULT"))
312 {
313 _exportPath = envVariables->variable("CAP_HISTOS_EXPORT_PATH");
314 }
315 else if (exportPath.BeginsWith("/"))
316 {
317 _exportPath = exportPath;
318 }
319 else
320 {
321 _exportPath = envVariables->variable("CAP_HISTOS_IMPORT_PATH");
322 _exportPath += "/";
323 _exportPath += exportPath;
324 }
325 }
326 if (_typeName.BeginsWith("PARTICLE_DB"))
327 {
328 if (importPath.BeginsWith("/"))
329 _importPath = importPath;
330 else
331 {
332 EnvironmentVariables * envVariables = EnvironmentVariables::environmentVariables();
333 _importPath= envVariables->variable("CAP_DATABASE_PATH");
334 _importPath += "/";
335 _importPath += importPath;
336 }
337 }
338 bool verbose = true;
339 if (reportInfo(__FUNCTION__))
340 {
341 verbose = true;
342 printCR();
343 printValue(createKey(_taskName,_typeName,"USE"), _use);
344 printValue(createKey(_taskName,_typeName,"CREATE"),_create);
345 printValue(createKey(_taskName,_typeName,"IMPORT"),_import);
346 printValue(createKey(_taskName,_typeName,"IMPORT:PATH"),_importPath);
347 printValue(createKey(_taskName,_typeName,"IMPORT:FILE_NAME"),_importFileName);
348 printValue(createKey(_taskName,_typeName,"EXPORT"),_export);
349 printValue(createKey(_taskName,_typeName,"EXPORT:PATH"),_exportPath);
350 printValue(createKey(_taskName,_typeName,"EXPORT:FILE_NAME"),_exportFileName);
351 printValue(createKey(_taskName,_typeName,"EXPORT:FORCE_REWRITE"),_forceRewrite);
352 printValue(createKey(_taskName,_typeName,"N"),nObjects);
353 }
354 MANAGED * object;
355
356 for (unsigned int k=0; k<nObjects; k++)
357 {
358 name = configuration.valueString(createKey(_taskName,_typeName,k,"NAME"));
359 title = configuration.valueString(createKey(_taskName,_typeName,k,"TITLE"));
360 owner = configuration.valueBool( createKey(_taskName,_typeName,k,"OWNER"));
361 if (verbose)
362 {
363 printValue(createKey(_taskName,_typeName,k,"NAME"),name);
364 printValue(createKey(_taskName,_typeName,k,"TITLE"),title);
365 printValue(createKey(_taskName,_typeName,k,"OWNER"),owner);
366 }
367 if (owner)
368 {
369 object = instantiateObject(name);
370 object->configure(_taskName,_typeName,configuration,k);
371 }
372 else
373 object = useObject(name);
374 }
375 }
