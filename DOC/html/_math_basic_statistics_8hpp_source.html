64template <typename T, typename Iterator, typename WeightIterator> T arithmeticMean(Iterator first, Iterator last, WeightIterator wfirst)
65{
66 T sum = 0;
67 T sumw = 0;
68 T weight = *wfirst;
69 while ( first != last )
70 {

140template <typename T, typename Iterator> T rms(Iterator first, Iterator last)
141{
142 T n = 0;
143 T tot = 0;
144 T x = 0;
145 T mean = arithmeticMean(first,last);
146 while ( first != last )
147 {
148 x = T(*first);
149 tot += (x - mean)*(x - mean);
150 ++first;
151 ++n;
152 }
153 double rms = (n > 1) ? sqrt(tot/(n-1)) : 0.0;
154 return rms;
155}

162template <typename T, typename Iterator, typename WeightIterator> T standardDeviation(Iterator first, Iterator last, WeightIterator wfirst)
163{
164 T tot = 0;
165 T sumw = 0;
166 T sumw2 = 0;
167 T x = 0;
168 T weight = 0;
169 T mean = arithmeticMean<T>(first,last,wfirst);
170 while ( first != last )
171 {
172 x = T(*first);
173 weight = *wfirst;
174 sumw += weight;
175 sumw2 += weight*weight;
176 tot += weight*(x-mean)*(x-mean);
177 ++first;
178 ++wfirst;
179 }
181 return std::sqrt(tot*sumw/(sumw*sumw-sumw2) );
