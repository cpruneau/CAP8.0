60/*
61double alf_array_index()
62This routine computes the index into a result_array[] corresponding
63to a given (l,m) using 'M' major indexing:
64
65(l,m) = (0,0) (1,0) (2,0) ... (L,0) (1,1) (2,1) ... (L,1) ... (L,L)
66
67index(l,m,L) = m*L - m(m-1)/2 + l
68*/
69
70INLINE_FUN
71size_t
72double alf_array_index(const size_t l, const size_t m, const size_t lmax)
73{
74 return (((m * (2 * lmax + 1 - m)) >> 1) + l);
75}
76
77/* return number of ALFs for a given lmax = (mmax+1)*(lmax+1) - mmax*(mmax+1)/2 */
78INLINE_FUN
79size_t
80double alf_nlm(const size_t lmax, const size_t mmax)
81{
82 const size_t M = (mmax > lmax) ? lmax : mmax;
83 return ((lmax + 1) * (M + 1) - ((M * (M + 1)) >> 1));
84}
