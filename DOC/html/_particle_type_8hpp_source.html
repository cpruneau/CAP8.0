124 int parseNumber(int flavorIndex, int content) const
125 {
126 if (content/1000000 > 0) throw Exception("More than 6 quark flavors??? value/1000000 > 0",__FUNCTION__);
127 int value = 0;
128 switch (flavorIndex)
129 {
130 case 0: value = int(content % 10); break;
131 case 1: value = int(content/10 % 10); break;
132 case 2: value = int(content/100 % 10); break;
133 case 3: value = int(content/1000 % 10); break;
134 case 4: value = int(content/10000 % 10); break;
135 case 5: value = int(content/100000 % 10); break;
136 default: throw Exception("Invalid flavor _index",__FUNCTION__);
137 }
138 return value;
139 }

161 int baryonNumber() const
162 {
163 int nd = quarkNumber(kDown);
164 int nu = quarkNumber(kUp);
165 int ns = quarkNumber(kStrange);
166 int nc = quarkNumber(kCharm);
167 int nb = quarkNumber(kBottom);
168 int nt = quarkNumber(kTop);
169 int nq = nd + nu + ns + nc + nb + nt;
170 int ndb = antiQuarkNumber(kDown);
171 int nub = antiQuarkNumber(kUp);
172 int nsb = antiQuarkNumber(kStrange);
173 int ncb = antiQuarkNumber(kCharm);
174 int nbb = antiQuarkNumber(kBottom);
175 int ntb = antiQuarkNumber(kTop);
176 int nqb = ndb + nub + nsb + ncb + nbb + ntb;
177 return (nq - nqb)/3;
178 }

187 int leptonNumber(unsigned int leptonIndex) const
188 {
189 int _charge = parseNumber(2*leptonIndex,_leptons);
190 if (_charge!=0) return _leptons >0 ? _charge : -_charge;
191 int neutrino = parseNumber(2*leptonIndex+1,_leptons);
192 if (neutrino!=0) return _leptons >0 ? neutrino : -neutrino;
193 return 0;
194 }
