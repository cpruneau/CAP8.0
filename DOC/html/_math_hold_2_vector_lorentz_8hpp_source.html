304 void setFromTXYZ(const T & t, const T & x, const T & y, const T & z)
305 {
306 switch (storageType)
307 {
308 default: throw MathException("Internal error","VectorLorentz<T>::setFromXYZ()");
309 break;
310
311 case 0:
312 v0 = t;
313 v1 = x;
314 v2 = y;
315 v3 = z;
316 break;
317
318 case 1:
319 {
320 T rhoSq = x*x + y*y;
321 v0 = t;
322 v1 = std::atan2(y,x);
323 v2 = std::atan2(std::sqrt(rhoSq),z);
324 v3 = std::sqrt(rhoSq + z*z);
325 break;
326 }
327
328 case 2:
329 {
330 v0 = t;
331 v1 = std::atan2(y,x);
332 v2 = std::sqrt(x*x + y*y);

334 break;

341 v0 = std::sqrt(mSq);
342 v1 = x;
343 v2 = y;
344 v3 = z;
345 break;
346 }
347 case 4:
348 {

350 if (mSq<0) throw MathException("t*t-x*x-y*y-z*z < 0 case 4","setFromTXYZ");
351 v0 = std::sqrt(mSq);
352 v1 = std::atan2(y,x);
