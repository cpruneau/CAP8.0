160TF1 *nNProfile(Double_t xSection, Double_t omega, Double_t G)
161{ // NN collisoin profile from https://arxiv.org/abs/1307.0636
162 if ((omega<0) || (omega>1))
163 return 0;
164 Double_t R2 = xSection/10./CAP::MATH::pi();
165 TF1 *nnprof = new TF1("nnprofgamma","[2]*(1-TMath::Gamma([0],[1]*x^2))",0,3);
166 nnprof->setParameters(1./omega,G/omega/R2,G);
167 return nnprof;
168}

204bool GlauberMC::calculateEvent(Double_t impactPar)
205{
206 // calc next event
207 if (!nucleonsA)
208 {
209 nucleonsA = nucleusA.getAllNucleons();
210 nucleonsA.size() = nucleusA.nNucleons();
211 for (Int_t i = 0; i<nucleonsA.size(); ++i) {
212 GlauberNucleon *nucleonA=(GlauberNucleon*)(nucleonsA->At(i));
213 nucleonA->setInNucleusA();
214 }
215 }
216
217 if (!nucleonsB)
218 {
219 nucleonsB = nucleusB.getAllNucleons();
220 nucleonsB.size() = nucleusB.nNucleons();
221 for (Int_t i = 0; i<nucleonsB.size(); ++i) {
222 GlauberNucleon *nucleonB=(GlauberNucleon*)(nucleonsB->At(i));
223 nucleonB->setInNucleusB();
224 }
225 }
226
227 Double_t xsecA[999] = {0};
228 Double_t xsecB[999] = {0};
229 if (fPTot) {
230 fXSectEvent = fPTot->getRandom();
231 if (fDoAAGG) {
232 for (Int_t i = 0; i<nucleonsA.size(); ++i)
233 xsecA[i] = fPTot->getRandom();
234 for (Int_t i = 0; i<nucleonsB.size(); ++i)
235 xsecB[i] = fPTot->getRandom();
236 }
237 }
238 else
239 fXSectEvent = fXSect;
240
241 // "ball" diameter = distance at which two balls interact
242 Double_t d2 = (Double_t)fXSectEvent/(CAP::MATH::pi()*10); // in fm^2
243 Double_t bh = TMath::Sqrt(d2*fHardFrac);
244 if (fNNProf) {
245 Double_t xmin=0,xmax=0;
246 fNNProf->getRange(xmin,xmax);
247 d2 = xmax*xmax;
248 }
249
250 eventStats.GlauberNucleon();
251 memset(fBC,0,sizeof(bool)*999*999);
252 Int_t nc=0,nh=0;
253 for (Int_t i = 0; i<nucleonsB.size(); ++i) {
254 GlauberNucleon *nucleonB=(GlauberNucleon*)(nucleonsB->At(i));
255 bool tB=nucleonB->nucleonType();
256 for (Int_t j = 0; j<nucleonsA.size(); ++j) {
257 GlauberNucleon *nucleonA=(GlauberNucleon*)(nucleonsA->At(j));
258 Double_t dx = nucleonB->getX()-nucleonA->getX();
259 Double_t dy = nucleonB->getY()-nucleonA->getY();
260 Double_t dij = dx*dx+dy*dy;
261 if (fDoAAGG && fPTot)
262 d2 = 0.5*(xsecA[j]+xsecB[i])/(CAP::MATH::pi()*10);
263 if (dij>d2)
264 continue;
265 Double_t bij = TMath::Sqrt(dij);
266 if (fNNProf) {
267 Double_t val = fNNProf->Eval(bij);
268 Double_t ran = gRandom->Uniform();
269 if (ran>val)
270 continue;
271 }
272 nucleonB->collide();
273 nucleonA->collide();
274 fBC[i][j] = 1;
275 eventStats.BNN += bij;
276 ++nc;
277 if (bij<bh)
278 ++nh;
279 bool tA=nucleonA->nucleonType();
280 if (tA!=tB)
281 ++eventStats.nCollisions_pn;
282 else if (tA==1)
283 ++eventStats.nCollisions_pp;
284 else
285 ++eventStats.nCollisions_nn;
286 if (nc==1) {
287 eventStats.X0 = (nucleonA->getX()+nucleonB->getX())/2;
288 eventStats.Y0 = (nucleonA->getY()+nucleonB->getY())/2;
289 }
290 }
291 }
292 eventStats.B = impactPar;
293 ++nTotalEvents;
294 if (nc>0) {
295 ++eventStatsents;
296 eventStats.nCollisions = nc;
297 eventStats.nHardCollisionsCollisions = nh;
298 eventStats.BNN /= nc;
299 return CalcResults(impactPar);
300 }
301 return kFALSE;
302}

304bool GlauberMC::CalcResults(Double_t impactPar)
305{
306 // calc results for the given event
307 Double_t sumW=0;
308 Double_t sumWA=0;
309 Double_t sumWB=0;
310
311 Double_t sinPhi[10] = {0};
312 Double_t cosPhi[10] = {0};
313 Double_t rn[10] = {0};
314
315 const Int_t kNc = fDoCore; // used later for core/corona
316
317 for (Int_t i = 0; i<nucleonsA.size(); ++i) {
318 GlauberNucleon *nucleonA=(GlauberNucleon*)(nucleonsA->At(i));
319 Double_t xA=nucleonA->getX();
320 Double_t yA=nucleonA->getY();
321 eventStats.meanXSystem += xA;
322 eventStats.meanYSystem += yA;
323 eventStats.meanXA += xA;
324 eventStats.meanYA += yA;
325 if (nucleonA->isWounded()) {
326 Double_t w = nucleonA->get2CWeight(f2Cx);
327 ++eventStats.nParticipants;
328 if (nucleonA->nBinaryCollisions()==1)
329 ++eventStats.nParticipants0;
330 ++eventStats.nParticipantsA;
331 sumW += w;
332 sumWA += w;
333 eventStats.meanX += xA * w;
334 eventStats.meanY += yA * w;
335 eventStats.meanX2 += xA * xA * w;
336 eventStats.meanY2 += yA * yA * w;
337 eventStats.meanXY += xA * yA * w;
338 }
339 }
340
341 for (Int_t i = 0; i<nucleonsB.size(); ++i) {
342 GlauberNucleon *nucleonB=(GlauberNucleon*)(nucleonsB->At(i));
343 Double_t xB=nucleonB->getX();
344 Double_t yB=nucleonB->getY();
345 eventStats.meanXSystem += xB;
346 eventStats.meanYSystem += yB;
347 eventStats.meanXB += xB;
348 eventStats.meanYB += yB;
349 if (nucleonB->isWounded()) {
350 Double_t w = nucleonB->get2CWeight(f2Cx);
351 ++eventStats.nParticipants;
352 if (nucleonB->nBinaryCollisions()==1)
353 ++eventStats.nParticipants0;
354 ++eventStats.nParticipantsB;
355 sumW += w;
356 sumWB += w;
357 eventStats.meanX += xB * w;
358 eventStats.meanY += yB * w;
359 eventStats.meanX2 += xB * xB * w;
360 eventStats.meanY2 += yB * yB * w;
361 eventStats.meanXY += xB * yB * w;
362 }
363 }
364 if (eventStats.nParticipants>0) {
365 eventStats.meanX /= sumW;
366 eventStats.meanY /= sumW;
367 eventStats.meanX2 /= sumW;
368 eventStats.meanY2 /= sumW;
369 eventStats.meanXY /= sumW;
370 } else {
371 eventStats.meanX = 0;
372 eventStats.meanY = 0;
373 eventStats.meanX2 = 0;
374 eventStats.meanY2 = 0;
375 eventStats.meanXY = 0;
376 }
377
378 if (nucleonsA.size()+nucleonsB.size()>0) {
379 eventStats.meanXSystem /= (nucleonsA.size() + nucleonsB.size());
380 eventStats.meanYSystem /= (nucleonsA.size() + nucleonsB.size());
381 } else {
382 eventStats.meanXSystem = 0;
383 eventStats.meanYSystem = 0;
384 }
385 if (nucleonsA.size()>0) {
386 eventStats.meanXA /= nucleonsA.size();
387 eventStats.meanYA /= nucleonsA.size();
388 } else {
389 eventStats.meanXA = 0;
390 eventStats.meanYA = 0;
391 }
392 if (nucleonsB.size()>0) {
393 eventStats.meanXB /= nucleonsB.size();
394 eventStats.meanYB /= nucleonsB.size();
395 } else {
396 eventStats.meanXB = 0;
397 eventStats.meanYB = 0;
398 }
399
400 eventStats.VarX = eventStats.meanX2-(eventStats.meanX*eventStats.meanX);
401 eventStats.VarY = eventStats.meanY2-(eventStats.meanY*eventStats.meanY);
402 eventStats.VarXY = eventStats.meanXY-eventStats.meanX*eventStats.meanY;
403 Double_t tmpa = eventStats.VarX*eventStats.VarY-eventStats.VarXY*eventStats.VarXY;
404 if (tmpa<0)
405 eventStats.areaW = -1;
406 else
407 eventStats.areaW = TMath::Sqrt(tmpa);
408
409 if (eventStats.nParticipants>0) {
410 // do full moments relative to meanX and meanY
411 for (Int_t n = 1; n<10; ++n) {
412 for (Int_t ia = 0; ia<nucleonsA.size(); ++ia) {
413 GlauberNucleon *nucleonA=(GlauberNucleon*)(nucleonsA->At(ia));
414 if (nucleonA->nBinaryCollisions()<=kNc)
415 continue;
416 Double_t xA=nucleonA->getX() - eventStats.meanX;
417 Double_t yA=nucleonA->getY() - eventStats.meanY;
418 Double_t r = TMath::Sqrt(xA*xA+yA*yA);
419 Double_t phi = TMath::ATan2(yA,xA);
420 Double_t w = n;
421 if (n==1)
422 w = 3; // use r^3 weighting for Ecc1/Psi1
423 Double_t rw = TMath::Power(r,w);
424 cosPhi[n] += rw*TMath::Cos(n*phi);
425 sinPhi[n] += rw*TMath::Sin(n*phi);
426 rn[n] += rw;
427 }
428 for (Int_t ib = 0; ib<nucleonsB.size(); ++ib) {
429 GlauberNucleon *nucleonB=(GlauberNucleon*)(nucleonsB->At(ib));
430 if (nucleonB->nBinaryCollisions()<=kNc)
431 continue;
432 Double_t xB=nucleonB->getX() - eventStats.meanX;
433 Double_t yB=nucleonB->getY() - eventStats.meanY;
434 Double_t r = TMath::Sqrt(xB*xB+yB*yB);
435 Double_t phi = TMath::ATan2(yB,xB);
436 Double_t w = n;
437 if (n==1)
438 w = 3; // use r^3 weighting for Ecc1/Psi1
439 Double_t rw = TMath::Power(r,w);
440 cosPhi[n] += rw*TMath::Cos(n*phi);
441 sinPhi[n] += rw*TMath::Sin(n*phi);
442 rn[n] += rw;
443 }
444 cosPhi[n] /= eventStats.nParticipants;
445 sinPhi[n] /= eventStats.nParticipants;
446 rn[n] /= eventStats.nParticipants;
447 if (rn[n]>0) {
448 fPsiN[n] = (TMath::ATan2(sinPhi[n],cosPhi[n]) + CAP::MATH::pi())/n;
449 fEccN[n] = TMath::Sqrt(sinPhi[n]*sinPhi[n]+cosPhi[n]*cosPhi[n])/rn[n];
450 } else {
451 fPsiN[n] = -1;
452 fEccN[n] = -1;
453 }
454 }
455 if (!kNc) { //silly test but useful to catch errors
456 Double_t t=TMath::Sqrt(TMath::Power(eventStats.VarY-eventStats.VarX,2)+4.*eventStats.VarXY*eventStats.VarXY)/(eventStats.VarY+eventStats.VarX)/fEccN[2];
457 if (t<0.99||t>1.01)
458 cout << "Error: Expected t=1 but found t=" << t << endl;
459 }
460 }
461
462 eventStats.B = impactPar;
463 eventStats.PhiA = nucleusA.phiRot();
464 eventStats.ThetaA = nucleusA.thetaRot();
465 eventStats.PhiB = nucleusB.phiRot();
466 eventStats.ThetaB = nucleusB.thetaRot();
467 eventStats.Psi1 = fPsiN[1];
468 eventStats.Ecc1 = fEccN[1];
469 eventStats.Psi2 = fPsiN[2];
470 eventStats.Ecc2 = fEccN[2];
471 eventStats.Psi3 = fPsiN[3];
472 eventStats.Ecc3 = fEccN[3];
473 eventStats.Psi4 = fPsiN[4];
474 eventStats.Ecc4 = fEccN[4];
475 eventStats.Psi5 = fPsiN[5];
476 eventStats.Ecc5 = fEccN[5];
477
478 if (fCalcArea) {
479 const Int_t nbins=200;
480 const Double_t ell=10;
481 const Double_t da=2*ell*2*ell/nbins/nbins;
482 const Double_t d2 = (Double_t)fXSectEvent/(CAP::MATH::pi()*10); // in fm^2
483 const Double_t r2 = d2/4.;
484 const Double_t mx = eventStats.meanX;
485 const Double_t my = eventStats.meanY;
486 TH2D areaA("hAreaA",";x (fm);y (fm)",nbins,-ell,ell,nbins,-ell,ell);
487 TH2D areaB("hAreaB",";x (fm);y (fm)",nbins,-ell,ell,nbins,-ell,ell);
488 for (Int_t i = 0; i<nucleonsA.size(); ++i) {
489 GlauberNucleon *nucleonA=(GlauberNucleon*)(nucleonsA->At(i));
490 if (!nucleonA->isWounded())
491 continue;
492 if (nucleonA->nBinaryCollisions()==kNc)
493 continue;
494 Double_t x = nucleonA->getX()-mx;
495 Double_t y = nucleonA->getY()-my;
496 for (Int_t xi=1; xi<=nbins; ++xi) {
497 for (Int_t yi=1; yi<=nbins; ++yi) {
498 Int_t bin = areaA.getBin(xi,yi);
499 Double_t val=areaA.getBinContent(bin);
500 if (val>0)
501 continue;
502 Double_t dx=x-areaA.getXaxis()->getBinCenter(xi);
503 Double_t dy=y-areaA.getYaxis()->getBinCenter(yi);
504 if (dx*dx+dy*dy<r2)
505 areaA.setBinContent(bin,1);
506 }
507 }
508 }
509 for (Int_t i = 0; i<nucleonsB.size(); ++i) {
510 GlauberNucleon *nucleonB=(GlauberNucleon*)(nucleonsB->At(i));
511 if (!nucleonB->isWounded())
512 continue;
513 if (nucleonB->nBinaryCollisions()==kNc)
514 continue;
515 Double_t x = nucleonB->getX()-mx;
516 Double_t y = nucleonB->getY()-my;
517 for (Int_t xi=1; xi<=nbins; ++xi) {
518 for (Int_t yi=1; yi<=nbins; ++yi) {
519 Int_t bin = areaB.getBin(xi,yi);
520 Double_t val=areaB.getBinContent(bin);
521 if (val>0)
522 continue;
523 Double_t dx=x-areaB.getXaxis()->getBinCenter(xi);
524 Double_t dy=y-areaB.getYaxis()->getBinCenter(yi);
525 if (dx*dx+dy*dy<r2)
526 areaB.setBinContent(bin,1);
527 }
528 }
529 }
530 Double_t overlap1=0;
531 Double_t overlap2=0;
532 for (Int_t xi=1; xi<=nbins; ++xi) {
533 for (Int_t yi=1; yi<=nbins; ++yi) {
534 Int_t bin = areaA.getBin(xi,yi);
535 Double_t vA=areaA.getBinContent(bin);
536 Double_t vB=areaB.getBinContent(bin);
537 if (vA>0&&vB>0)
538 ++overlap1;
539 if (vA>0||vB>0)
540 ++overlap2;
541 }
542 }
543 eventStats.AreaO = overlap1*da;
544 eventStats.AreaA = overlap2*da;
545 }
546
547 if (fCalcLength) {
548 const Double_t krhs = TMath::Sqrt(fXSectEvent/40./CAP::MATH::pi());
549 const Double_t ksg = krhs/TMath::Sqrt(5);
550 const Double_t kDL = 0.1;
551 TF1 rad("rad","2*pi/[0]/[0]*TMath::Exp(-x*x/(2.*[0]*[0]))",0.0,5*ksg);
552 rad.setParameter(0,ksg);
553 const Double_t minval = rad.Eval(5*ksg);
554 eventStats.Phi0 = gRandom->Uniform(0,TMath::TwoPi());
555 Double_t kcphi0 = TMath::Cos(eventStats.Phi0);
556 Double_t ksphi0 = TMath::Sin(eventStats.Phi0);
557 Double_t x = eventStats.X0;
558 Double_t y = eventStats.Y0;
559 Double_t i0a = 0;
560 Double_t i1a = 0;
561 Double_t l = 0;
562 Double_t val = CalcDens(rad,x,y);
563 while (val>minval) {
564 x += kDL * kcphi0;
565 y += kDL * ksphi0;
566 i0a += val;
567 i1a += l*val;
568 l+=kDL;
569 val = CalcDens(rad,x,y);
570 }
571 eventStats.Length = 2*i1a/i0a;
572 }
573
574 if (eventStats.nParticipants > fMaxnParticipantsFound)
575 fMaxnParticipantsFound = eventStats.nParticipants;
576
577 return kTRUE;
578}

580Double_t GlauberMC::CalcDens(TF1 & prof, Double_t xval, Double_t yval) const
581{
582 Double_t rmin=0,rmax=0;
583 prof.getRange(rmin,rmax);
584 Double_t r2max = rmax*rmax;
585 Double_t ret = 0;
586 for (Int_t i = 0; i<nucleonsA.size(); ++i) {
587 GlauberNucleon *nucleonA=(GlauberNucleon*)(nucleonsA->At(i));
588 if (!nucleonA->isWounded())
589 continue;
590 Double_t x = nucleonA->getX();
591 Double_t y = nucleonA->getY();
592 Double_t r2=(xval-x)*(xval-x)+(yval-y)*(yval-y);
593 if (r2>r2max)
594 continue;
595 ret += prof.Eval(TMath::Sqrt(r2));
596 }
597 for (Int_t i = 0; i<nucleonsB.size(); ++i) {
598 GlauberNucleon *nucleonB=(GlauberNucleon*)(nucleonsB->At(i));
599 if (!nucleonB->isWounded())
600 continue;
601 Double_t x = nucleonB->getX();
602 Double_t y = nucleonB->getY();
603 Double_t r2=(xval-x)*(xval-x)+(yval-y)*(yval-y);
604 if (r2>r2max)
605 continue;
606 ret += prof.Eval(TMath::Sqrt(r2));
607 }
608 return ret;
609}
