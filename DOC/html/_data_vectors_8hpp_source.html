29 inline double statErrorProduct(double & v1, double & e1, double & v2, double & e2)
30 {
31 // do not use relative errors and avoid divisions
32 double ep1 = v1*e2;
33 double ep2 = v2*e1;
34 return sqrt(ep1*ep1+ep2*ep2);
35 }

37 inline double statErrorRatio(double & v1, double & e1, double & v2, double & e2)
38 {
39 // do not use relative errors and minimize number of divisions
40 if (v2==0) return 1.0E100;
41 double er2 = e2*v1/v2;
42 return sqrt(e1*e1+er2*er2)/v2;
43 }
