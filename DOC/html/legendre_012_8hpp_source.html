101/* P_l^m(x) m >= 0; l >= m; |x| <= 1.0
102 *
103 * Note that this function grows combinatorially with l.
104 * Therefore we can easily generate an overflow for l larger
105 * than about 150.
106 *
107 * There is no trouble for small m, but when m and l are both large,
108 * then there will be trouble. Rather than allow overflows, these
109 * functions refuse to calculate when they can sense that l and m are
110 * too big.
111 *
112 * If you really want to calculate a spherical harmonic, then DO NOT
113 * use this. Instead use legendre_sphPlm() below, which uses a similar
114 * recursion, but with the normalized functions.
115 *
116 * exceptions: throw MathDomainException("",__FUNCTION__), GSL_EOVRFLW
117 */

252/* The following spherical functions are specializations
253 * of Legendre functions which give the regular eigenfunctions
254 * of the Laplacian on a 3-dimensional hyperbolic space.
255 * Of particular interest is the flat limit, which is
256 * Flat-Lim := {lambda->Inf, eta->0, lambda*eta fixed}.
257 */
258
259/* Zeroth radial eigenfunction of the Laplacian on the
260 * 3-dimensional hyperbolic space.
261 *
262 * legendre_H3d_0(lambda,eta) := sin(lambda*eta)/(lambda*sinh(eta))
263 *
264 * Normalization:
265 * Flat-Lim legendre_H3d_0(lambda,eta) = j_0(lambda*eta)
266 *
267 * eta >= 0.0
268 * exceptions: throw MathDomainException("",__FUNCTION__)
269 */
270double legendre_H3d_0(double lambda, double eta);
271double legendre_H3d_0(double lambda, double eta);
272
273
274/* First radial eigenfunction of the Laplacian on the
275 * 3-dimensional hyperbolic space.
276 *
277 * legendre_H3d_1(lambda,eta) :=
278 * 1/sqrt(lambda^2 + 1) sin(lam eta)/(lam sinh(eta))
279 * (coth(eta) - lambda cot(lambda*eta))
280 *
281 * Normalization:
282 * Flat-Lim legendre_H3d_1(lambda,eta) = j_1(lambda*eta)
283 *
284 * eta >= 0.0
285 * exceptions: throw MathDomainException("",__FUNCTION__)
286 */
287double legendre_H3d_1(double lambda, double eta);
288double legendre_H3d_1(double lambda, double eta);
